mysql comandos basicos

use <database>; = vai para um banco de dados

create database <nome>; = cria um banco e dados

create table <nome> (<nome_coluna1> <tipo_coluna1>, ..., <nome_colunaN> <tipo_colunaN>); = cria uma tabela com as colunas com os tipos especificados

show tables; = exibe as tabelas do banco de dados

show columns from <tabela>; = exibe a estrutura da tabela
